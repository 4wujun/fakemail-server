86  	@Test
87  	public void testSearchByPost() throws Exception {
88  		final EmailSearchCommand command =
89  				new EmailSearchCommand();
90  		command.setSenderId(1L);
91  		command.setRecipient("test@foo.org");
92  		command.setSentSince(new Date());
93  		command.setSentBefore(new Date());
94  		command.setOffset(10);
95  		command.setLimit(20);
96  		final EmailSearchReport report =
97  				new EmailSearchReport();
98  		report.setId(1L);
99  		report.setSender("foor@bar.com");
100 		report.setRecipient("test@foo.org");
101 		report.setSentDate(new Date());
102 		report.setSubject("Test mail controller");
103 		final PagedList<EmailSearchReport> list =
104 				new PagedList<EmailSearchReport>();
105 		list.setData(Arrays.asList(report));
106 		list.setTotal(5);
107 
108 		when(emailService.search(command)).thenReturn(list);
109 
110 		final MockHttpServletRequestBuilder postBuilder =
111 				post("/api/mail");
112 		postBuilder.accept(MediaType.APPLICATION_JSON_UTF8)
113 			.contentType(MediaType.APPLICATION_JSON_UTF8)
114 			.content(TestUtils.convertObjectToJson(command));
115 		mockMvc.perform(postBuilder)
116 			.andExpect(status().isOk())
117 			.andExpect(content().contentType(MediaType.APPLICATION_JSON_UTF8))
118 			.andExpect(jsonPath("total", is(5)))
119 			.andExpect(jsonPath("data", hasSize(1)))
120 			.andExpect(jsonPath("data[0].id", is(report.getId().intValue())))
121 			.andExpect(jsonPath("data[0].recipient",
122 					is(report.getRecipient())))
123 			.andExpect(jsonPath("data[0].sentDate",
124 					is(report.getSentDate().getTime())))
125 			.andExpect(jsonPath("data[0].subject", is(report.getSubject())))
126 			;
127 		verify(emailService, times(1)).search(command);
128 		verifyNoMoreInteractions(emailService);
129 	}
130

133 		MockitoAnnotations.initMocks(this);
134 	}
