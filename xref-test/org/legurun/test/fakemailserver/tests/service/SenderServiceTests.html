43  import static org.mockito.Mockito.*;
44  
45  /**
46   * Sender service tests.
47   * @author patrice
48   * @since 2017
49   */
50  @RunWith(SpringJUnit4ClassRunner.class)
51  @ContextConfiguration(classes = { RootConfig.class, RepositoryConfig.class })
52  @TestPropertySource(value = "classpath:application-test-h2.properties")
53  public class SenderServiceTests {
54  	/**
55  	 * Email service to test.
56  	 */
57  	@Autowired
58  	@InjectMocks
59  	private ISenderService senderService;
60  
61  	@Mock
62  	private ISenderDao senderDao;
63  
64  	@Test
65  	public void testList() {
66  		final List<Sender> senders = new ArrayList<Sender>();
67  		final Sender sender = new Sender();
68  		sender.setAddress("foo@bar.com");
69  		senders.add(sender);
70  		when(senderDao.list()).thenReturn(senders);
71  
72  		final List<Sender> result = senderService.list();
73  		assertEquals("Result list must contains only one result",
74  				senders, result);
75  		verify(senderDao, times(1)).list();
76  	}
77  
78  
79  	@Test
80  	public void testGet() {
81  		final Sender sender = new Sender();
82  		when(senderDao.get(1L)).thenReturn(null);
83  		when(senderDao.get(2L)).thenReturn(sender);
84  
85  		final Sender resultNull = senderService.get(1L);
86  		assertNull("Result must be null", resultNull);
87  
88  		final Sender resultNotNull = senderService.get(2L);
89  		assertNotNull("Result must be not null", resultNotNull);
90  		verify(senderDao, times(1)).get(eq(1L));
91  		verify(senderDao, times(1)).get(eq(2L));
92  	}
93  
94  	@Before
95  	public void initMockito() {
96  		MockitoAnnotations.initMocks(this);
97  		ReflectionTestUtils.setField(senderService, "senderDao", senderDao);
98  	}
99  }
